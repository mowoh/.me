    • Task 1: Basic Class Hierarchy (Inheritance)

    Inheritance in OOP depicts a parent (basee class) offprings (dreived classes)relationship.
    The following steps are taken to showcase this concept in task 1:
  • Define a base class Vehicle with brand and fuel_type as instance attributes (use __init__).
  • Create a subclass Car that inherits from Vehicle and adds num_doors as an additional attribute.
  • Ensure the Car class calls the parent class's __init__ method (using super() in Python)
    • .

    Solution

    
class Vehicle:
    def __init__(self, brand: str, fuel_type: str, year: int, color: str) -> None:
        self.brand = brand
        self.fuel_type = fuel_type
        self.year = year
        self.color = color
    def drive(self) -> None:
        print(f"Driving {self.brand} {self.fuel_type} {self.year} {self.color}")

class Car(Vehicle):
    def __init__(self, brand: str, fuel_type: str, year: int, color: str, num_doors: int) -> None:
        super().__init__(brand, fuel_type, year, color)
        self.num_doors = num_doors

    def drive(self) -> None:
        print(f"Driving {self.brand} {self.fuel_type} {self.year} {self.color} with {self.num_doors} doors")

car = Car("Toyota", "Gasoline", 2024, "Red", 4)
car.drive()

# Extras
class Motorcycle(Vehicle):
    def __init__(self, brand: str, fuel_type: str, year: int, color: str, engine_size: int) -> None:

        self.engine_size = engine_size

    def drive(self) -> None:
        print(f"Driving {self.brand} {self.fuel_type} {self.year} {self.color} with a {self.engine_size} engine")

motorcycle = Motorcycle("Honda", "Gasoline", 2024, "Blue", 1000)

    Task 2: Polymorphism with Methods

    Define an abstract Shape class with an abstract method area() (use ABC and abstractmethod in Python). Create Circle and Rectangle subclasses that inherit from Shape. Implement area() in each subclass (for Circle, use πr²; for Rectangle, use length * width).

    Solution

    from abc import ABC, abstractmethod
import math
class Shape(ABC):
    @abstractmethod
    def area(self) -> float:
        pass

class Circle(Shape):
    def __init__(self, radius: float) -> None:
        self.radius = radius

    def area(self) -> float:
        return math.pi * self.radius ** 2

class Rectangle(Shape):
    def __init__(self, length: float, width: float) -> None:
        self.length = length
        self.width = width

    def area(self) -> float:
        return self.length * self.width

circle  = Circle(3.7);
print(f"Area of circle: {circle.area()}")

    Task 3: Encapsulation with Access Control

    Define BankAccount with a private attribute __balance (use double underscore in Python). Provide public methods: deposit(amount) to add to __balance. withdraw(amount) to deduct from __balance (check for sufficient funds). Use getter methods if balance access is needed (e.g., get_balance())..

    Solution

    
class BankAccount:
    def __init__(self, initial_balance: float = 0.0) -> None:
        self.__balance = initial_balance

    def deposit(self, amount: float) -> None:
        if amount > 0:
            self.__balance += amount
        else:
            raise ValueError("Amount must be positive")

    def withdraw(self, amount: float) -> None:
        if self.__balance >= amount:
            self.__balance -= amount
        else:
            raise ValueError("Insufficient funds")

    def get_balance(self) -> float:
        return self.__balance

    Task 4: Abstraction with Base Class

    Create an abstract Animal class with an abstract method make_sound(). Implement subclasses Dog and Cat, each overriding make_sound() to return "Woof!" and "Meow!" respectively. Use ABC and abstractmethod decorators in Python.

    Solution

    from abc import ABC, abstractmethod

class Animal(ABC):
    @abstractmethod
    def make_sound(self) -> str:
        pass

class Cat(Animal):
    def make_sound(self) -> str:
        return "Meow!"

class Dog(Animal):
    def make_sound(self) -> str:
        return "Woof!"
class Bird(Animal):
    def make_sound(self) -> str:
        return "Tweet!"
class Fish(Animal):
    def make_sound(self) -> str:
        return "Blub!"
class Snake(Animal):
    def make_sound(self) -> str:
        return "Hiss!"
class Horse(Animal):
    def make_sound(self) -> str:
        return "Neigh!"
class Pig(Animal):
    def make_sound(self) -> str:
        return "Oink!"

    Task 5: Constructor and Destructor

    Define a Person class with __init__(self, name) to set the name attribute. Add a destructor __del__(self) that prints a farewell message (e.g., "Goodbye, {name}!"). Test by creating and deleting an instance (use del explicitly or let it go out of scope).

    Solution

    

class Person:
    def __init__(self, first_name:str, last_name:str) -> None:
        self.first_name = first_name
        self.last_name = last_name

    def __del__(self) -> None:

    Task 6: Class Methods and Static Methods

    Define a Car class with a class method create_car(brand, model) that returns a new Car instance. Add a static method calculate_mileage(distance, fuel_used) to calculate mileage (distance / fuel_used). Test by creating a car instance and using both class and static methods.

    Solution

    
  
class Car:
    def __init__(self, brand: str, model: str) -> None:
        self.brand = brand
        self.model = model

    @classmethod
    def create_car(cls, brand: str, model: str) -> "Car":
        return cls(brand, model)

    @staticmethod
    def calculate_mileage(distance: float, fuel_used: float) -> float:
        return distance / fuel_used

car = Car.create_car("Toyota", "Corolla")
print(f"Car brand: {car.brand}, Car model: {car.model}")
print(f"Mileage: {Car.calculate_mileage(100, 10)}")
